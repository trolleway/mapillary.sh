
ls -d *.JPG | parallel --bar bash ./filename2date.sh {} 2020-05-24
time exiftool -overwrite_original "-FileModifyDate>CreateDate" *.JPG && exiftool -overwrite_original "-CreateDate>DateTimeOriginal" *.JPG
--forward

mkdir mapillary_copy
ls -d *.JPG | parallel --bar convert  -crop 4002x2551+606+309 -quality 85% "{}" "mapillary_copy/{}"
mapillary_tools process --advanced --import_path mapillary_copy --user_name trolleway  --verbose --rerun --interpolate_directions  --offset_angle 0 --cutoff_distance 100 --cutoff_time 60 --overwrite_EXIF_direction_tag
mapillary_tools upload --import_path mapillary_copy --skip_subfolders --number_threads 2 --max_attempts 10 --advanced
echo 'ok'
